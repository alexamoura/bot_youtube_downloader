#!/usr/bin/env python3
"""
Shopee Watermark Remover - FFmpeg
Remove marca d'√°gua de v√≠deos da Shopee ap√≥s download
"""

import subprocess
import os
import logging

LOG = logging.getLogger(__name__)


class WatermarkRemover:
    """
    Remove marca d'√°gua de v√≠deos usando FFmpeg
    """
    
    # Posi√ß√µes comuns da marca d'√°gua da Shopee
    SHOPEE_POSITIONS = {
        'top_left': {'x': 10, 'y': 10, 'w': 200, 'h': 50},
        'top_right': {'x': 'iw-210', 'y': 10, 'w': 200, 'h': 50},
        'bottom_left': {'x': 10, 'y': 'ih-60', 'w': 200, 'h': 50},
        'bottom_right': {'x': 'iw-210', 'y': 'ih-60', 'w': 200, 'h': 50},
        'top_center': {'x': '(iw-200)/2', 'y': 10, 'w': 200, 'h': 50},
    }
    
    def __init__(self):
        # Verifica se FFmpeg est√° dispon√≠vel
        self.ffmpeg_available = self._check_ffmpeg()
    
    def _check_ffmpeg(self) -> bool:
        """Verifica se FFmpeg est√° instalado"""
        try:
            subprocess.run(
                ['ffmpeg', '-version'],
                capture_output=True,
                check=True
            )
            LOG.info("‚úÖ FFmpeg dispon√≠vel")
            return True
        except (subprocess.CalledProcessError, FileNotFoundError):
            LOG.warning("‚ö†Ô∏è FFmpeg n√£o encontrado")
            return False
    
    def detect_watermark_position(self, video_path: str) -> dict:
        """
        Tenta detectar posi√ß√£o da marca d'√°gua automaticamente
        Retorna melhor estimativa
        """
        # Por enquanto, retorna posi√ß√£o mais comum da Shopee
        # TODO: Implementar detec√ß√£o autom√°tica com OpenCV
        
        LOG.info("üìç Usando posi√ß√£o padr√£o: canto inferior direito")
        return self.SHOPEE_POSITIONS['bottom_right']
    
    def remove_delogo(
        self, 
        input_path: str, 
        output_path: str,
        x: int = 10,
        y: int = 10,
        w: int = 200,
        h: int = 50
    ) -> bool:
        """
        Remove marca d'√°gua usando filtro delogo do FFmpeg
        
        Args:
            input_path: Caminho do v√≠deo original
            output_path: Caminho para salvar v√≠deo limpo
            x, y: Posi√ß√£o da marca (canto superior esquerdo)
            w, h: Largura e altura da √°rea da marca
        
        Returns:
            True se sucesso, False caso contr√°rio
        """
        if not self.ffmpeg_available:
            LOG.error("‚ùå FFmpeg n√£o dispon√≠vel")
            return False
        
        try:
            LOG.info("üé¨ Removendo marca d'√°gua com delogo...")
            LOG.info(f"   Posi√ß√£o: x={x}, y={y}, w={w}, h={h}")
            
            # Converte express√µes para valores
            if isinstance(x, str):
                x_expr = x
            else:
                x_expr = str(x)
            
            if isinstance(y, str):
                y_expr = y
            else:
                y_expr = str(y)
            
            cmd = [
                'ffmpeg',
                '-i', input_path,
                '-vf', f'delogo=x={x_expr}:y={y_expr}:w={w}:h={h}:show=0',
                '-c:a', 'copy',  # Mant√©m √°udio original
                '-y',  # Sobrescreve se existir
                output_path
            ]
            
            result = subprocess.run(
                cmd,
                capture_output=True,
                text=True,
                check=True
            )
            
            LOG.info("‚úÖ Marca d'√°gua removida com sucesso!")
            return True
            
        except subprocess.CalledProcessError as e:
            LOG.error(f"‚ùå Erro no FFmpeg: {e.stderr}")
            return False
        except Exception as e:
            LOG.error(f"‚ùå Erro inesperado: {e}")
            return False
    
    def remove_blur(
        self,
        input_path: str,
        output_path: str,
        x: int = 10,
        y: int = 10,
        w: int = 200,
        h: int = 50
    ) -> bool:
        """
        Borra a √°rea da marca d'√°gua (alternativa ao delogo)
        Fica mais natural em alguns casos
        """
        if not self.ffmpeg_available:
            return False
        
        try:
            LOG.info("üå´Ô∏è Borrando √°rea da marca d'√°gua...")
            
            # Filtro de blur na √°rea espec√≠fica
            blur_filter = (
                f"[0:v]crop={w}:{h}:{x}:{y},boxblur=10[blurred];"
                f"[0:v][blurred]overlay={x}:{y}"
            )
            
            cmd = [
                'ffmpeg',
                '-i', input_path,
                '-filter_complex', blur_filter,
                '-c:a', 'copy',
                '-y',
                output_path
            ]
            
            subprocess.run(cmd, capture_output=True, check=True)
            
            LOG.info("‚úÖ √Årea borrada com sucesso!")
            return True
            
        except Exception as e:
            LOG.error(f"‚ùå Erro ao borrar: {e}")
            return False
    
    def remove_auto(
        self,
        input_path: str,
        output_path: str,
        position: str = 'bottom_right'
    ) -> bool:
        """
        Remove marca d'√°gua automaticamente usando posi√ß√£o pr√©-definida
        
        Args:
            position: 'top_left', 'top_right', 'bottom_left', 'bottom_right', 'top_center'
        """
        if position not in self.SHOPEE_POSITIONS:
            LOG.error(f"‚ùå Posi√ß√£o inv√°lida: {position}")
            return False
        
        pos = self.SHOPEE_POSITIONS[position]
        
        LOG.info(f"üéØ Removendo marca d'√°gua da posi√ß√£o: {position}")
        
        return self.remove_delogo(
            input_path,
            output_path,
            x=pos['x'],
            y=pos['y'],
            w=pos['w'],
            h=pos['h']
        )
    
    def try_all_positions(
        self,
        input_path: str,
        output_dir: str
    ) -> dict:
        """
        Tenta remover marca d'√°gua de todas as posi√ß√µes
        √ötil para descobrir onde est√° a marca
        
        Gera um v√≠deo para cada posi√ß√£o:
        - video_top_left.mp4
        - video_top_right.mp4
        - etc.
        """
        results = {}
        
        for position, coords in self.SHOPEE_POSITIONS.items():
            output_path = os.path.join(
                output_dir,
                f"video_{position}.mp4"
            )
            
            LOG.info(f"üß™ Testando posi√ß√£o: {position}")
            
            success = self.remove_delogo(
                input_path,
                output_path,
                **coords
            )
            
            results[position] = {
                'success': success,
                'path': output_path if success else None
            }
        
        return results

def remove_shopee_watermark(
    video_path: str,
    position: str = 'bottom_right'
) -> str:
    """
    Fun√ß√£o simples para remover marca d'√°gua da Shopee
    
    Args:
        video_path: Caminho do v√≠deo
        position: Posi√ß√£o da marca ('bottom_right' padr√£o)
    
    Returns:
        Caminho do v√≠deo sem marca ou None se falhar
    """
    remover = WatermarkRemover()
    
    if not remover.ffmpeg_available:
        LOG.warning("‚ö†Ô∏è FFmpeg n√£o dispon√≠vel - v√≠deo mant√©m marca")
        return video_path
    
    # Cria nome do arquivo de sa√≠da
    base, ext = os.path.splitext(video_path)
    output_path = f"{base}_no_watermark{ext}"
    
    # Remove marca
    success = remover.remove_auto(video_path, output_path, position)
    
    if success:
        # Remove arquivo original e renomeia
        try:
            os.remove(video_path)
            os.rename(output_path, video_path)
            LOG.info("‚úÖ V√≠deo original substitu√≠do por vers√£o sem marca")
            return video_path
        except:
            LOG.warning("‚ö†Ô∏è N√£o foi poss√≠vel substituir arquivo original")
            return output_path
    else:
        LOG.warning("‚ö†Ô∏è Falha ao remover marca - mantendo original")
        return video_path


def remove_watermark_custom(
    video_path: str,
    x: int,
    y: int,
    w: int,
    h: int
) -> str:
    """
    Remove marca d'√°gua com coordenadas personalizadas
    """
    remover = WatermarkRemover()
    
    if not remover.ffmpeg_available:
        return video_path
    
    base, ext = os.path.splitext(video_path)
    output_path = f"{base}_no_watermark{ext}"
    
    success = remover.remove_delogo(video_path, output_path, x, y, w, h)
    
    if success:
        try:
            os.remove(video_path)
            os.rename(output_path, video_path)
            return video_path
        except:
            return output_path
    else:
        return video_path
